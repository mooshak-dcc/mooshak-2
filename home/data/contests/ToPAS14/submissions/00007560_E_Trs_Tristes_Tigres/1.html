#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char dic[1000][21], aCodificar[21];

void printInt(int N) {
    int i;

    for(i=0; i<N; i++) {
        if(strcmp(dic[i], aCodificar) == 0) {
            printf("%d", i+1);
            break;
        }
    }

    return;
}

int main()
{

    for(i=0; i<N; i++) {
        scanf("%s", dic[i]);
    }

    scanf("%d", &op);
    while(op != 0) {
        if(op == 1) {
            scanf("%d", &c);
            for(i=0; i<c; i++) {

    }

    return 0;
}
